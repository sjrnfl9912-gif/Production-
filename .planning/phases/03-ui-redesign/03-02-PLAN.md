**수정 2 — loadMyRecords() 함수를 날짜 그룹화로 교체**

기존 `loadMyRecords()` 함수 전체를 아래로 교체:

```javascript
function loadMyRecords(){
  const w=document.getElementById('myWorkerSelect').value,area=document.getElementById('myRecordsArea');
  if(!w){area.innerHTML='<div class="empty-row">작업자를 선택해주세요.</div>';return}
  const mine=records.filter(r=>r.worker===w).sort((a,b)=>new Date(b.date)-new Date(a.date));
  if(!mine.length){area.innerHTML='<div class="empty-row">데이터가 없습니다.</div>';return}
  const byDate={};
  mine.forEach(r=>{if(!byDate[r.date])byDate[r.date]=[];byDate[r.date].push(r)});
  let h='';
  Object.keys(byDate).sort((a,b)=>b.localeCompare(a)).forEach(date=>{
    const ok=isWithin7Days(date);
    h+=`<div class="records-date-group">`;
    h+=`<div class="records-date-header">${fmtDate(date)}${!ok?' <span class="text-xs-muted">(수정 불가)</span>':''}</div>`;
    h+=`<table class="edit-table"><thead><tr><th>구분</th><th>작업명</th><th>수량</th><th>비고</th><th style="width:90px">관리</th></tr></thead><tbody>`;
    byDate[date].forEach(r=>{
      const badge=r.type==='메인작업'?'badge-main':r.type==='밑작업'?'badge-sub':'badge-etc';
      h+=`<tr><td data-label="구분"><span class="badge ${badge}">${r.type}</span></td>
        <td data-label="작업명">${r.taskName}</td><td data-label="수량" class="fw-700">${r.quantity} EA</td>
        <td data-label="비고" class="text-muted">${r.note||'-'}</td>
        <td>${ok?`<button class="btn btn-outline btn-sm" onclick="openEditModal(${r.id},false)">수정</button> <button class="btn btn-danger btn-sm" onclick="deleteRecord(${r.id},false)">삭제</button>`:'<span class="text-xs-muted">기한 만료</span>'}</td></tr>`;
    });
    h+='</tbody></table></div>';
  });
  area.innerHTML=h;
}
```
